#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=1;i<=t;i++)
    {
        int n;
        cin>>n;
        int arr[n];
        int count=0;
        for(int i=0;i<n;i++)
        {
            cin>>arr[i];
        }
        sort(arr,arr+n);
        if(n>1){
        int ans = 1;
        count=1;
        for(int i=1;i<n;i++)
        {
            if(arr[i-1]!=arr[i])
            {
                count++;
            }
            ans += count ;
        }
        cout<<"Case #"<<t<<": "<<ans<<endl;
        }
        else
        {
            int ans = 1;
            cout<<"Case #"<<t<<": "<<ans<<endl;
        }
    }
}
